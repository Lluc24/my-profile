% Business card — terminal/hacker aesthetic, light theme.
% Compile with:  lualatex card.tex   (requires fontspec; LuaLaTeX or XeLaTeX only)
%
% All geometry is in millimetres. TikZ is set to x=1mm,y=1mm below so every
% bare coordinate number in the pictures IS a millimetre — no unit suffixes,
% no "in", no mixed-unit arithmetic bugs.
\documentclass[tikz,border=0mm]{standalone}
\usepackage{graphicx}

% ---------------------------------------------------------------------------
% Font selection: monospace only, with a real fallback chain. fontspec's
% \IfFontExistsTF checks the system font database (fontconfig) at compile
% time, so this never silently drops to a serif — the last resort is still
% a mono face (DejaVu Sans Mono ships with virtually every TeX/Linux install).
% ---------------------------------------------------------------------------
\IfFontExistsTF{JetBrains Mono}
  {\setmainfont{JetBrains Mono}}
  {\IfFontExistsTF{Fira Code}
    {\setmainfont{Fira Code}}
    {\IfFontExistsTF{IBM Plex Mono}
      {\setmainfont{IBM Plex Mono}}
      {\setmainfont{DejaVu Sans Mono}}}}

% Separate semi-bold face for the window title, mirroring the same fallback
% chain. If no semibold-named static instance exists anywhere in the chain,
% \semiboldfont just falls back to \bfseries of whichever main font loaded
% above (still bold, never a different family, never a serif).
\IfFontExistsTF{JetBrains Mono SemiBold}
  {\newfontfamily\semiboldfont{JetBrains Mono SemiBold}}
  {\IfFontExistsTF{Fira Code SemiBold}
    {\newfontfamily\semiboldfont{Fira Code SemiBold}}
    {\IfFontExistsTF{IBM Plex Mono SemiBold}
      {\newfontfamily\semiboldfont{IBM Plex Mono SemiBold}}
      {\let\semiboldfont\bfseries}}}

% Ligature-free instance of the same main font, for spots where the font's
% own "calt"/"liga" tables would otherwise silently rewrite plain ASCII --
% e.g. turning a typed "->" into a rendered arrow, or fusing a run of "-"
% characters into one long dash. IMPORTANT: \addfontfeature mid-paragraph
% does NOT reliably turn "calt" off (verified empirically -- a scoped
% \addfontfeature left contextual ligatures untouched while a fresh
% \fontspec/\newfontfamily declaration did not), so this needs its own
% \newfontfamily rather than an inline feature toggle.
\IfFontExistsTF{JetBrains Mono}
  {\newfontfamily\noligfont[RawFeature={-calt;-liga}]{JetBrains Mono}}
  {\IfFontExistsTF{Fira Code}
    {\newfontfamily\noligfont[RawFeature={-calt;-liga}]{Fira Code}}
    {\IfFontExistsTF{IBM Plex Mono}
      {\newfontfamily\noligfont[RawFeature={-calt;-liga}]{IBM Plex Mono}}
      {\newfontfamily\noligfont[RawFeature={-calt;-liga}]{DejaVu Sans Mono}}}}

% ---------------------------------------------------------------------------
% Palette — light terminal theme, colors pulled from GitHub Light (chosen
% over Solarized Light / Catppuccin Latte because its accent hues are the
% only ones of the three that clear 4.5:1 against a light background — see
% the contrast table below, measured with the standard WCAG formula).
%
% cardbg is plain #FFFFFF -- deliberately NOT an off-white/warm tint. Print
% shops often bill "background color" as full-bleed ink coverage regardless
% of hue; #FFFFFF prints as no ink at all on white stock, i.e. the bare
% paper color, so this card doesn't pay for a custom near-white.
%
%   color                  hex       vs #FFFFFF card bg   used for
%   -----------------------------------------------------------------------
%   cardbg      #FFFFFF      —                    card background (bare paper)
%   headerbg    #EBEBEB      1.09 vs cardbg        chrome bar fill (decorative)
%   headerborder#D0D0D0      —                     chrome bar 1px bottom rule
%   bodycolor   #1F2328    15.80                    body text, window title
%   mutedtext   #656D76     5.25                    secondary/dim lines
%   accent      #1A7F37     5.08                    primary accent (green)
%   ruleline    #C8C8C8      —                      fastfetch separator (decorative)
%   cropcolor   #B4B4B4      —                      faint crop-mark gray (decorative)
%
% accent and mutedtext both clear 4.5:1 against the card background; the
% two backgrounds (cardbg/headerbg) and the two rule colors are decorative
% fills/lines, not text, so they aren't held to the text contrast target.
% bluecard (the pass-1 legacy accent) is dropped: the fastfetch spec calls
% for exactly one accent color for every key, so a second accent had no use
% left in this layout.
% ---------------------------------------------------------------------------
\definecolor{cardbg}{RGB}{255,255,255}
\definecolor{headerbg}{RGB}{235,235,235}
\definecolor{headerborder}{RGB}{208,208,208}
\definecolor{bodycolor}{RGB}{31,35,40}
\definecolor{mutedtext}{RGB}{101,109,118}
\definecolor{accent}{RGB}{26,127,55}
\definecolor{ruleline}{RGB}{200,200,200}
\definecolor{cropcolor}{RGB}{180,180,180}
% winctrlbg/winctrlicon sampled directly from a real GNOME headerbar
% screenshot (Screenshot from 2026-07-19 18-34-48.png): each of the three
% window-control glyphs sits on its own subtle circular "chip", a shade
% darker than the headerbar itself -- decorative, not held to text contrast.
\definecolor{winctrlbg}{RGB}{218,218,218}
\definecolor{winctrlicon}{RGB}{61,61,61}

% ---------------------------------------------------------------------------
% Crop marks toggle. The canvas below IS the bleed area (there's no extra
% sheet margin beyond it), so these are short faint ticks sitting in the
% bleed band at each trim corner rather than full press-standard crop marks
% extended onto a larger sheet. Flip to \showcropmarksfalse to hide them.
% ---------------------------------------------------------------------------
\newif\ifshowcropmarks
\showcropmarkstrue

% Draws the 8 corner ticks for the currently-defined \cardwidth/\cardheight/
% \bleed/\fullwidth/\fullheight, in the OUTER (full-canvas) coordinate frame,
% i.e. call this before/after the \bleed-shifted scope, not inside it.
\newcommand{\drawcropmarks}{%
  \ifshowcropmarks
    \draw[cropcolor, line width=0.08mm]
      % bottom-left trim corner (\bleed,\bleed)
      (0,\bleed) -- (\bleed-0.5,\bleed)
      (\bleed,0) -- (\bleed,\bleed-0.5)
      % bottom-right trim corner (\fullwidth-\bleed,\bleed)
      (\fullwidth,\bleed) -- (\fullwidth-\bleed+0.5,\bleed)
      (\fullwidth-\bleed,0) -- (\fullwidth-\bleed,\bleed-0.5)
      % top-left trim corner
      (0,\fullheight-\bleed) -- (\bleed-0.5,\fullheight-\bleed)
      (\bleed,\fullheight) -- (\bleed,\fullheight-\bleed+0.5)
      % top-right trim corner
      (\fullwidth,\fullheight-\bleed) -- (\fullwidth-\bleed+0.5,\fullheight-\bleed)
      (\fullwidth-\bleed,\fullheight) -- (\fullwidth-\bleed,\fullheight-\bleed+0.5);
  \fi
}

% ---------------------------------------------------------------------------
% Adwaita-style headerbar, shared by both faces so the chrome only has to be
% tuned in one place. Takes the titlebar text (e.g. "lluc@card: ~/front") as
% its argument. Must be called from inside the \bleed-shifted scope, after
% the card-body background fill, so the bar paints on top of it.
% Reads \cardwidth/\cardheight/\bleed/\barheight/\safemargin from the caller.
% ---------------------------------------------------------------------------
\newcommand{\headerbar}[1]{%
  % bar fill, bleeding to the top/side edges
  \fill[headerbg] (-\bleed,\cardheight-\barheight) rectangle (\cardwidth+\bleed,\cardheight+\bleed);
  % 1px bottom border separating the bar from the card body
  \draw[headerborder, line width=0.25mm]
    (-\bleed,\cardheight-\barheight) -- (\cardwidth+\bleed,\cardheight-\barheight);
  % centered semi-bold window title
  \node[anchor=center] at (\cardwidth/2,\cardheight-\barheight/2) {%
    \fontsize{7}{8}\selectfont\semiboldfont\color{bodycolor} #1%
  };
  % three window controls, right-aligned in GTK order (minimize, maximize,
  % close), each on its own circular "chip" -- this whole treatment
  % (colors, chip size, icon weight, pitch) was matched directly against a
  % real GNOME headerbar screenshot rather than guessed. Minimize/maximize
  % are drawn as plain TikZ shapes -- a short dash, a small square outline
  % -- rather than font glyphs, since JetBrains Mono's coverage of
  % window-control symbols isn't guaranteed; close stays \texttimes
  % (multiplication sign), a standard text symbol already confirmed to
  % render correctly.
  \def\winctrly{\cardheight-\barheight/2}
  \def\winctrlsize{1.6}     % dash length / square side, mm
  \def\winctrlradius{2.5}   % chip circle radius, mm
  \def\winctrlpitch{6}      % spacing between chip centers, mm
  \def\closex{\cardwidth-\safemargin-\winctrlradius}
  \def\maximizex{\closex-\winctrlpitch}
  \def\minimizex{\maximizex-\winctrlpitch}
  % chip backgrounds, drawn first so the icons sit on top of them
  \fill[winctrlbg] (\minimizex,\winctrly) circle (\winctrlradius);
  \fill[winctrlbg] (\maximizex,\winctrly) circle (\winctrlradius);
  \fill[winctrlbg] (\closex,\winctrly) circle (\winctrlradius);
  % minimize -- short horizontal dash
  \draw[winctrlicon, line width=0.3mm]
    (\minimizex-\winctrlsize/2,\winctrly) -- (\minimizex+\winctrlsize/2,\winctrly);
  % maximize -- small square outline
  \draw[winctrlicon, line width=0.25mm]
    (\maximizex-\winctrlsize/2,\winctrly-\winctrlsize/2) rectangle (\maximizex+\winctrlsize/2,\winctrly+\winctrlsize/2);
  % close -- sized up from the original 7.5pt (and set bold) so its visual
  % weight matches the dash/square, which are drawn at a fixed mm stroke
  % width rather than scaling with a font size
  \node[anchor=center] at (\closex,\winctrly) {%
    \fontsize{11}{12}\selectfont\bfseries\color{winctrlicon} \texttimes%
  };
}

    % "at (4,\contentcentery)" a y-component that itself starts with "(" --
    % pgf's coordinate-list parser sees that nested ")" as closing the

    % text. Pre-computing the number sidesteps that parser entirely.
    \pgfmathsetmacro{\contentcentery}{(\bodytop+\safemargin)/2}

    % curly right-double-quote glyph, which would quietly wreck this art
    % exactly the way the task warned about. \raggedright keeps LaTeX from

    % \contentcentery, same as the info column below.
